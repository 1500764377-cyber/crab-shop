<script>
const products = [
  {id:1,name:"公蟹 3.5两",price:68},
  {id:2,name:"母蟹 2.5两",price:78},
  {id:3,name:"8只礼盒装",price:588},
  {id:4,name:"蟹醋+工具套装",price:19.9}
];

const goodsEl = document.getElementById("goods");
products.forEach(p=>{
  const div = document.createElement("div");
  div.className = "card";
  div.innerHTML = `
  <img src="https://picsum.photos/seed/crab${p.id}/400/300">
  <div class="info">
    <div class="name">${p.name}</div>
    <div class="price">¥${p.price}</div>
    <button class="btn" onclick="addCart(${p.id})">加入购物车</button>
  </div>`;
  goodsEl.appendChild(div);
});

let cart = JSON.parse(localStorage.getItem("cart")) || [];

function addCart(id){
  const item = products.find(p=>p.id==id);
  const find = cart.find(c=>c.id==id);
  if(find) find.num++;
  else cart.push({...item, num:1});
  saveCart();
  alert("已加入购物车");
}

function saveCart(){
  localStorage.setItem("cart", JSON.stringify(cart));
  renderCart();
}

function renderCart(){
  const cartList = document.getElementById("cartList");
  cartList.innerHTML = "";
  let total = 0;
  cart.forEach(item=>{
    const money = item.price * item.num;
    total += money;
    cartList.innerHTML += `
    <div class="cart-item">
      <span>${item.name} ×${item.num}</span>
      <span>¥${money.toFixed(2)}</span>
    </div>`;
  });
  document.getElementById("total").innerText = total.toFixed(2);
}

function clearCart(){
  cart = [];
  saveCart();
}

function submitOrder(){
  const name = document.getElementById("name").value.trim();
  const phone = document.getElementById("phone").value.trim();
  const address = document.getElementById("address").value.trim();
  const total = document.getElementById("total").innerText;

  if(!name || !phone || !address){
    alert("请填写完整收货信息");
    return;
  }
  if(cart.length==0){
    alert("购物车不能为空");
    return;
  }

  const order = {
    id: Date.now(),
    name, phone, address,
    total, goods: cart,
    time: new Date().toLocaleString(),
    payStatus: "待支付"
  };

  let orders = JSON.parse(localStorage.getItem("orders")) || [];
  orders.unshift(order);
  localStorage.setItem("orders", JSON.stringify(orders));

  location.href = `pay.html?orderId=${order.id}&total=${total}`;
}

renderCart();
